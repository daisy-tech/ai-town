# 本地开发方案迁移指南

## 🎯 目标

将AI Town从Convex云服务迁移到本地开发环境，适合学习和测试。

## 📋 方案概述

### 当前架构（Convex）
```
Frontend → Convex Cloud → Vector Database → LLM API
```

### 本地架构
```
Frontend → Local Backend → SQLite/JSON → LLM API (可选)
```

## 🛠️ 实施方案

### 方案A: 简化版（推荐用于学习）

**特点**:
- 不使用向量数据库
- 使用JSON文件存储
- 简化记忆检索（基于关键词）
- 保留核心AI对话功能

**实施步骤**:

#### 1. 创建本地存储模块

创建 `src/localBackend/storage.ts`:

```typescript
import fs from 'fs';
import path from 'path';

interface Memory {
  id: string;
  playerId: string;
  playerName: string;
  description: string;
  importance: number;
  type: 'conversation' | 'reflection' | 'relationship';
  createdAt: number;
  lastAccess: number;
  metadata?: any;
}

class LocalStorage {
  private dataDir = './data';
  private memoriesFile = path.join(this.dataDir, 'memories.json');
  private memories: Memory[] = [];

  constructor() {
    this.ensureDataDir();
    this.loadMemories();
  }

  private ensureDataDir() {
    if (!fs.existsSync(this.dataDir)) {
      fs.mkdirSync(this.dataDir, { recursive: true });
    }
  }

  private loadMemories() {
    try {
      if (fs.existsSync(this.memoriesFile)) {
        const data = fs.readFileSync(this.memoriesFile, 'utf-8');
        this.memories = JSON.parse(data);
      }
    } catch (error) {
      console.error('Failed to load memories:', error);
      this.memories = [];
    }
  }

  private saveMemories() {
    try {
      fs.writeFileSync(
        this.memoriesFile,
        JSON.stringify(this.memories, null, 2),
        'utf-8'
      );
    } catch (error) {
      console.error('Failed to save memories:', error);
    }
  }

  // 添加记忆
  addMemory(memory: Omit<Memory, 'id' | 'createdAt' | 'lastAccess'>): Memory {
    const newMemory: Memory = {
      ...memory,
      id: `mem_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
      createdAt: Date.now(),
      lastAccess: Date.now(),
    };

    this.memories.push(newMemory);
    this.saveMemories();
    return newMemory;
  }

  // 简单的关键词搜索（替代向量搜索）
  searchMemories(playerId: string, keywords: string[], limit: number = 5): Memory[] {
    const now = Date.now();

    // 过滤该角色的记忆
    let results = this.memories.filter(m => m.playerId === playerId);

    // 如果有关键词，按匹配度排序
    if (keywords.length > 0) {
      results = results.map(memory => {
        let score = 0;
        const lowerDesc = memory.description.toLowerCase();

        keywords.forEach(keyword => {
          if (lowerDesc.includes(keyword.toLowerCase())) {
            score += 10;
          }
        });

        // 时效性衰减
        const hoursSinceAccess = (now - memory.lastAccess) / (1000 * 60 * 60);
        const recencyScore = Math.pow(0.99, Math.floor(hoursSinceAccess));

        // 综合评分
        const totalScore = score + memory.importance + recencyScore * 5;

        return { memory, score: totalScore };
      });

      // 排序
      results.sort((a, b) => b.score - a.score);

      // 返回记忆对象
      results = results.map(r => r.memory);
    } else {
      // 没有关键词，按时间和重要性排序
      results.sort((a, b) => {
        const scoreA = a.importance + (a.lastAccess / 1000000000);
        const scoreB = b.importance + (b.lastAccess / 1000000000);
        return scoreB - scoreA;
      });
    }

    // 更新访问时间
    const selected = results.slice(0, limit);
    selected.forEach(m => {
      m.lastAccess = now;
    });
    this.saveMemories();

    return selected;
  }

  // 获取所有记忆
  getAllMemories(playerId?: string): Memory[] {
    if (playerId) {
      return this.memories.filter(m => m.playerId === playerId);
    }
    return this.memories;
  }

  // 清空所有记忆
  clearAll() {
    this.memories = [];
    this.saveMemories();
  }

  // 获取统计信息
  getStats() {
    const total = this.memories.length;
    const byType = {
      conversation: this.memories.filter(m => m.type === 'conversation').length,
      reflection: this.memories.filter(m => m.type === 'reflection').length,
      relationship: this.memories.filter(m => m.type === 'relationship').length,
    };

    return { total, byType };
  }
}

export const storage = new LocalStorage();
export type { Memory };
```

#### 2. 创建简化的LLM接口

创建 `src/localBackend/llm.ts`:

```typescript
// 如果有OpenAI API Key，可以使用真实API
// 否则使用模拟响应

interface LLMConfig {
  apiKey?: string;
  useRealAPI: boolean;
}

const config: LLMConfig = {
  apiKey: process.env.OPENAI_API_KEY,
  useRealAPI: !!process.env.OPENAI_API_KEY,
};

export async function generateResponse(
  playerName: string,
  otherPlayerName: string,
  context: string,
  memories: string[]
): Promise<string> {
  if (config.useRealAPI && config.apiKey) {
    // 使用真实的OpenAI API
    const response = await fetch('https://api.openai.com/v1/chat/completions', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'Authorization': `Bearer ${config.apiKey}`,
      },
      body: JSON.stringify({
        model: 'gpt-3.5-turbo',
        messages: [
          {
            role: 'system',
            content: `你是${playerName}。相关记忆：\n${memories.join('\n')}`,
          },
          {
            role: 'user',
            content: `${context}\n\n请用中文简短回复（50字以内）。`,
          },
        ],
        max_tokens: 100,
      }),
    });

    const data = await response.json();
    return data.choices[0].message.content;
  } else {
    // 使用模拟响应（用于测试）
    const responses = [
      `你好${otherPlayerName}！很高兴见到你。`,
      `我们一起努力守护地球吧！`,
      `最近的训练进展不错。`,
      `团队合作真的很重要。`,
      `让我们继续加油！`,
    ];

    return responses[Math.floor(Math.random() * responses.length)];
  }
}

export function extractKeywords(text: string): string[] {
  // 简单的中文关键词提取
  const keywords = text
    .replace(/[，。！？、]/g, ' ')
    .split(' ')
    .filter(word => word.length >= 2);

  return keywords.slice(0, 5); // 最多5个关键词
}
```

#### 3. 修改游戏逻辑

创建 `src/localBackend/gameAdapter.ts`:

```typescript
import { storage, Memory } from './storage';
import { generateResponse, extractKeywords } from './llm';

// 适配器：连接本地存储和游戏逻辑
export class LocalGameAdapter {
  // 记录对话
  async recordConversation(
    playerId: string,
    playerName: string,
    otherPlayerId: string,
    otherPlayerName: string,
    messages: { author: string; text: string }[]
  ): Promise<void> {
    // 生成对话摘要
    const summary = messages
      .map(m => `${m.author}: ${m.text}`)
      .join('\n');

    const description = `与${otherPlayerName}的对话：${summary.substring(0, 200)}`;

    // 简单的重要性评分（基于长度和关键词）
    const importance = Math.min(9, Math.floor(messages.length / 2) + 4);

    // 保存记忆
    storage.addMemory({
      playerId,
      playerName,
      description,
      importance,
      type: 'conversation',
      metadata: {
        otherPlayerId,
        otherPlayerName,
        messageCount: messages.length,
      },
    });

    console.log(`💾 保存了${playerName}的对话记忆`);
  }

  // 生成回复
  async generateMessage(
    playerId: string,
    playerName: string,
    otherPlayerName: string,
    conversationContext: string
  ): Promise<string> {
    // 提取关键词
    const keywords = extractKeywords(conversationContext + ' ' + otherPlayerName);

    // 搜索相关记忆
    const memories = storage.searchMemories(playerId, keywords, 3);

    // 格式化记忆
    const memoryTexts = memories.map(m => `- ${m.description.substring(0, 100)}`);

    // 生成回复
    const response = await generateResponse(
      playerName,
      otherPlayerName,
      conversationContext,
      memoryTexts
    );

    return response;
  }

  // 获取记忆统计
  getStats() {
    return storage.getStats();
  }

  // 清空数据
  clearAll() {
    storage.clearAll();
    console.log('🗑️ 已清空所有数据');
  }
}

export const gameAdapter = new LocalGameAdapter();
```

#### 2. 数据库Schema

```sql
CREATE TABLE memories (
  id TEXT PRIMARY KEY,
  player_id TEXT NOT NULL,
  player_name TEXT NOT NULL,
  description TEXT NOT NULL,
  importance INTEGER NOT NULL,
  type TEXT NOT NULL,
  created_at INTEGER NOT NULL,
  last_access INTEGER NOT NULL,
  metadata TEXT
);

CREATE INDEX idx_player_id ON memories(player_id);
CREATE INDEX idx_type ON memories(type);
```

#### 3. 向量存储（使用hnswlib）

```typescript
import { HierarchicalNSW } from 'hnswlib-node';

class VectorStore {
  private index: HierarchicalNSW;
  private dimension = 384; // 使用更小的模型

  constructor() {
    this.index = new HierarchicalNSW('l2', this.dimension);
    this.index.initIndex(10000); // 最多1万条记忆
  }

  addVector(id: number, vector: number[]) {
    this.index.addPoint(vector, id);
  }

  search(queryVector: number[], k: number = 5): number[] {
    const result = this.index.searchKnn(queryVector, k);
    return result.neighbors;
  }
}
```
